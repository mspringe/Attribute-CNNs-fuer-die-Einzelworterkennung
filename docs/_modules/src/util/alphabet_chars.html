[docs]def dset_chars(dset: dl.GWDataSet):
    """
    Returns unique characters of the data set

    :param dset: data set
    :return: the data sets unique characters
    """
    words = list(dset.words)
    all_chars = ''.join(words)
    set_chars = ''.join(sorted(set(all_chars)))
    return set_chars
